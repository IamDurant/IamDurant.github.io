```java
class UnionFind {
    int[] s;
    int[] rank;

    public UnionFind(int size) {
        s = new int[size];
        rank = new int[size];
        for (int i = 0; i < size; i++) {
            s[i] = i;
        }
    }

    public int find(int x) {
        // path compression
        if (s[x] != x) {
            s[x] = find(s[x]);
        }

        return s[x];
    }

    public void union(int i, int j) {
        int iBoss = find(i);
        int jBoss = find(j);

        if (iBoss != jBoss) {
            if(iBoss < jBoss) {
                s[jBoss] = iBoss;
            } else {
                s[iBoss] = jBoss;
            }
        }
    }
}
```

#### 完全版

- 有 union by rank
```java
class DisjointSet {
    int[] s;
    int[] rank;

    public DisjointSet(int size) {
        s = new int[size];
        rank = new int[size];
        for (int i = 0; i < size; i++) {
            s[i] = i;
        }
    }

    public int find(int x) {
        // path compression
        if (s[x] != x) {
            s[x] = find(s[x]);
        }

        return s[x];
    }

    public void union(int i, int j) {
        int iBoss = find(i);
        int jBoss = find(j);

        if (iBoss != jBoss) {
            // union by rank
            if (rank[iBoss] < rank[jBoss]) {
                s[iBoss] = jBoss;
            } else if (rank[iBoss] > rank[jBoss]) {
                s[jBoss] = iBoss;
            } else {
                s[iBoss] = jBoss;
                rank[jBoss]++;
            }
        }
    }

    @Override
    public String toString() {
        return "DisjointSet{" +
            "s=" + Arrays.toString(s) +
            '}';
    }
}
```
